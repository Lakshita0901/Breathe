function div(kg: number, divisor: number, unit: string): string {
  const n = Math.round(kg / divisor);
  return n <= 0 ? ZERO : `= ${n} ${unit}`;
}
 
function mul(kg: number, multiplier: number, unit: string): string {
  const n = Math.round(kg * multiplier);
  return n <= 0 ? ZERO : `= ${n} ${unit}`;
}
